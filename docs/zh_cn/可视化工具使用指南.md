## 一、可视化模块概述

### 1.1 模块结构

```python
"""
czsc.utils.plotting 模块结构：

plotting/
├── __init__.py          # 统一导出接口
├── common.py            # 通用常量和辅助函数
├── backtest.py          # 回测图表
├── weight.py            # 权重分析图表
└── kline.py             # K线图表
"""
```

### 1.2 快速导入

```python
# 推荐：从统一接口导入
from czsc.utils.plotting import (
    # 回测图表
    plot_cumulative_returns,
    plot_backtest_stats,
    plot_monthly_heatmap,
    plot_drawdown_analysis,

    # K线图表
    KlineChart,
    plot_czsc_chart,

    # 分析图表
    plot_daily_return_distribution,
    plot_colored_table,

    # 样式常量
    COLOR_DRAWDOWN,
    COLOR_RETURN,
)

# 或者直接从子模块导入
from czsc.utils.plotting.backtest import plot_cumulative_returns
from czsc.utils.plotting.kline import KlineChart
```

## 二、K线图表

### 2.1 基础K线图

```python
from czsc.utils.plotting import KlineChart
from czsc.core import CZSC

# 创建K线图表
kc = KlineChart(
    n_rows=3,          # 子图行数
    height=800,        # 图表高度
    title="000001 分析图表"
)

# 添加K线
kc.add_kline(
    czsc_obj,          # CZSC分析对象
    name="000001",     # 名称
    row=1              # 子图位置
)

# 显示图表
kc.fig.show()
```

### 2.2 添加技术指标

```python
# 添加均线
kc.add_sma(
    czsc_obj,
    period=5,
    name="MA5",
    row=1,
    color="red"
)

kc.add_sma(
    czsc_obj,
    period=20,
    name="MA20",
    row=1,
    color="blue"
)

# 添加成交量
kc.add_vol(
    czsc_obj,
    row=2
)

# 添加MACD
kc.add_macd(
    czsc_obj,
    row=3
)

kc.fig.show()
```

### 2.3 添加技术指标

```python
# KlineChart 支持的绑定方法：
# - add_kline: 添加K线图
# - add_vol: 添加成交量
# - add_sma: 添加均线
# - add_macd: 添加MACD
# - add_indicator: 添加自定义指标线
# - add_marker_indicator: 添加标记指标
# - add_scatter_indicator: 添加散点指标
# - add_bar_indicator: 添加柱状指标

# 示例：添加均线
kc.add_sma(czsc_obj, period=5, row=1)
kc.add_sma(czsc_obj, period=20, row=1)

# 添加自定义指标
kc.add_indicator(
    dt_values,      # 时间序列
    indicator_values, # 指标值
    name="RSI",
    row=2
)

kc.fig.show()
```

> **注意**：`KlineChart` 不提供 `add_fx`、`add_bi`、`add_zs` 等缠论元素的
> 独立绑定方法。如需绘制完整的缠论分析图（包含笔、分型、中枢标注），
> 请使用 `plot_czsc_chart` 函数（通过 `from czsc import plot_czsc_chart` 导入），
> 该函数内部使用 `add_scatter_indicator` 实现缠论元素的绑定。

### 2.4 多品种对比

```python
# 创建多品种图表
kc = KlineChart(n_rows=2, height=600)

# 添加第一个品种
kc.add_kline(czsc_obj_1, name="000001", row=1)
kc.add_sma(czsc_obj_1, period=5, row=1)  # 添加均线辅助分析

# 添加第二个品种
kc.add_kline(czsc_obj_2, name="000002", row=2)
kc.add_sma(czsc_obj_2, period=5, row=2)

kc.fig.show()
```

## 三、回测分析图表

### 3.1 累计收益曲线

```python
from czsc.utils.plotting import plot_cumulative_returns
import pandas as pd

# 准备日收益数据
dret = pd.DataFrame({
    'dt': pd.date_range('20230101', '20231231', freq='D'),
    '策略': strategy_returns,      # 策略日收益
    '基准': benchmark_returns,     # 基准日收益
}).set_index('dt')

# 绘制累计收益曲线
fig = plot_cumulative_returns(
    dret,
    title="策略累计收益对比",
    template="plotly",       # 或 "plotly_dark"
    to_html=False           # 返回 Figure 对象
)

fig.show()
```

```python
    template="plotly"
```

## 六、图表样式定制

### 6.1 使用颜色常量

```python
from czsc.utils.plotting import (
    COLOR_DRAWDOWN,     # 回撤颜色
    COLOR_RETURN,       # 收益颜色
    COLOR_ANNO_GRAY,    # 标注灰色
    COLOR_ANNO_RED,     # 标注红色
    COLOR_BORDER,       # 边框颜色
    COLOR_HEADER_BG,    # 表头背景色
)

print(f"回撤颜色: {COLOR_DRAWDOWN}")
print(f"收益颜色: {COLOR_RETURN}")
```

### 6.2 自定义样式

```python
# 使用自定义样式
fig = plot_cumulative_returns(
    dret,
    title="自定义样式图表",
    template="plotly_dark",   # 深色主题
    height=600,               # 自定义高度
    width=1000,               # 自定义宽度
    color_discrete_sequence=["#FF0000", "#00FF00", "#0000FF"]
)
```

### 6.3 添加注释

```python

fig = plot_cumulative_returns(dret)

    x="2023-06-01",
    y=0,
    text="重要事件",

fig.show()
```

## 七、图表导出

### 7.1 导出为HTML

```python
# 导出为交互式HTML
html_str = plot_cumulative_returns(
    dret,
    to_html=True  # 返回HTML字符串
)

# 保存到文件
with open("backtest.html", "w", encoding="utf-8") as f:
    f.write(html_str)
```

### 7.2 导出为图片

```python
# 导出为静态图片（需要 kaleido）
fig = plot_cumulative_returns(dret, to_html=False)

# 导出为PNG
fig.write_image("backtest.png", width=1200, height=600, scale=2)

# 导出为PDF
fig.write_image("backtest.pdf", width=1200, height=600)
```

### 7.3 导出为SVG

```python
# 导出为SVG（矢量图，可编辑）
fig.write_image("backtest.svg", width=1200, height=600)
```

## 八、常见问题

### Q1: 中文显示乱码怎么办？

A: 配置字体：

```python
import plotly.graph_objects as go

# 设置中文字体
fig = plot_cumulative_returns(dret)
fig.update_layout(
    font=dict(
        family="SimHei",  # 黑体
        size=12
    )
)
```

### Q2: 图表太大或太小？

A: 调整尺寸：

```python
fig = plot_cumulative_returns(
    dret,
    height=800,   # 像素高度
    width=1200    # 像素宽度
)
```

### Q3: 如何保存图表配置？

A: 使用模板：

```python
# 保存配置
config = {
    'template': 'plotly_dark',
    'height': 800,
    'width': 1200,
    'colors': ['#FF0000', '#00FF00']
}

# 应用配置
fig = plot_cumulative_returns(dret, **config)
```

## 九、最佳实践

### 9.1 图表设计原则

```python
"""
图表设计原则：

1. 清晰性
   - 去除无关元素
   - 突出关键信息

2. 一致性
   - 使用统一的颜色方案
   - 保持图表风格一致

3. 可读性
   - 合适的字体大小
   - 清晰的标签和图例

4. 信息密度
   - 避免过度拥挤
   - 合理的子图布局
"""
```

### 9.2 图表使用场景

```python
"""
场景与图表对应：

1. 策略展示
   → 累计收益曲线
   → 回测统计概览

2. 绩效分析
   → 月度热力图
   → 回撤分析
   → 收益分布

3. 技术分析
   → K线图表
   → 指标图表

4. 风险分析
   → 回撤分析
   → 权重分布
"""
```
