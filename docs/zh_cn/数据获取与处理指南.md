## 一、数据接口概述

### 1.1 DataClient 统一接口

```python
from czsc.utils import DataClient

# 创建数据客户端
dc = DataClient(
    token="your_token",              # API凭证
    url="http://api.tushare.pro",    # API地址
    timeout=300,                     # 请求超时时间
    verbose=True                     # 显示详细日志
)

# 获取K线数据
df = dc.get_kline(
    symbol="000001",    # 标的代码
    freq="日线",        # K线周期
    sdt="20230101",     # 开始日期
    edt="20231231",     # 结束日期
    fq="前复权"         # 复权方式
)
```

### 1.2 支持的数据源

| 数据源 | 标识 | 支持市场 | 数据类型 | 说明 |
|--------|------|----------|----------|------|
| **Tushare** | `ts` | A股 | 日线、分钟线 | 需要token |
| **聚宽** | `jq` | A股 | 日线、分钟线 | 需要账户 |
| **天勤** | `tq` | 期货 | Tick、分钟线 | 免费实时 |
| **CCXT** | `ccxt` | 数字货币 | Tick、K线 | 多交易所 |
| **本地文件** | `file` | - | CSV/Feather | 本地数据 |

## 三、数据格式转换

### 3.1 DataFrame 转 RawBar

```python
from czsc.core import format_standard_kline, Freq

# 标准数据格式
required_columns = [
    'dt',      # 日期时间
    'open',    # 开盘价
    'close',   # 收盘价
    'high',    # 最高价
    'low',     # 最低价
    'vol',     # 成交量
    'amount'   # 成交额
]

# 确保数据列名正确
df = df.rename(columns={
    'trade_date': 'dt',
    'volume': 'vol',
    'amount': 'amount'
})

# 转换时间格式
df['dt'] = pd.to_datetime(df['dt'])

# 转换为 RawBar 对象
bars = format_standard_kline(df, freq=Freq.F30)

# 查看结果
print(f"转换后K线数量: {len(bars)}")
print(f"第一根K线: {bars[0]}")
```

### 3.2 RawBar 转 DataFrame

```python
from czsc.core import RawBar
import pandas as pd

def bars_to_dataframe(bars: list) -> pd.DataFrame:
    """将 RawBar 列表转换为 DataFrame

    参数：
        bars: RawBar 对象列表

    返回：
        DataFrame
    """
    data = [{
        'symbol': bar.symbol,
        'dt': bar.dt,
        'id': bar.id,
        'open': bar.open,
        'close': bar.close,
        'high': bar.high,
        'low': bar.low,
        'vol': bar.vol,
        'amount': bar.amount,
    } for bar in bars]

    return pd.DataFrame(data)

# 使用
df = bars_to_dataframe(bars)
```

## 四、BarGenerator K线合成

### 4.1 基础用法

```python
from czsc.core import BarGenerator

# 创建K线生成器
bg = BarGenerator(
    base_freq='1分钟',    # 基础周期
    freqs=['5分钟', '15分钟', '30分钟', '日线']  # 目标周期
)

# 输入基础周期K线
for bar in bars_1m:
    bg.update(bar)

# 获取合成后的K线
bars_5m = bg.bars['5分钟']
bars_30m = bg.bars['30分钟']
bars_daily = bg.bars['日线']

print(f"1分钟: {len(bars_1m)} 根")
print(f"5分钟: {len(bars_5m)} 根")
print(f"30分钟: {len(bars_30m)} 根")
print(f"日线: {len(bars_daily)} 根")
```

## 六、数据校验与清洗

### 6.1 数据校验

```python
from czsc.utils.data.validators import validate_kline_data

# 校验K线数据
is_valid, errors = validate_kline_data(df)

if not is_valid:
    print("数据校验失败:")
    for error in errors:
        print(f"  - {error}")
else:
    print("数据校验通过")

# 常见错误类型：
# - 缺失值
# - 异常值（价格为负或零）
# - 时间不连续
# - 高低点关系错误
```

### 6.2 数据清洗

```python
import pandas as pd
import numpy as np

def clean_kline_data(df):
    """清洗K线数据

    参数：
        df: 原始K线数据

    返回：
        清洗后的数据
    """
    df = df.copy()

    # 1. 处理缺失值
    df = df.dropna(subset=['open', 'close', 'high', 'low'])

    # 2. 处理异常值
    # 价格必须为正
    for col in ['open', 'close', 'high', 'low']:
        df = df[df[col] > 0]

    # 成交量不能为负
    df = df[df['vol'] >= 0]

    # 3. 修正高低点关系
    df['high'] = df[['open', 'close', 'high']].max(axis=1)
    df['low'] = df[['open', 'close', 'low']].min(axis=1)

    # 4. 去除重复
    df = df.drop_duplicates(subset=['dt'])

    # 5. 排序
    df = df.sort_values('dt').reset_index(drop=True)

    return df

# 使用
df_clean = clean_kline_data(df_raw)
```

### 6.3 异常检测

```python
def detect_anomalies(df):
    """检测数据异常

    返回异常记录
    """
    anomalies = []

    # 检测价格跳空（超过20%）
    df['prev_close'] = df['close'].shift(1)
    df['gap'] = abs(df['open'] / df['prev_close'] - 1)
    gap_anomalies = df[df['gap'] > 0.2]

    # 检测成交异常（超过均值10倍）
    vol_mean = df['vol'].mean()
    vol_anomalies = df[df['vol'] > vol_mean * 10]

    # 检测停牌（无成交）
    zero_vol = df[df['vol'] == 0]

    return {
        '跳空异常': gap_anomalies,
        '成交异常': vol_anomalies,
        '停牌': zero_vol
    }
```

## 八、常见问题

### Q1: 数据缺失怎么办？

A: 使用前向填充或线性插值：

```python
# 检测缺失日期
all_dates = pd.date_range(start=df['dt'].min(), end=df['dt'].max(), freq='D')
missing_dates = set(all_dates) - set(df['dt'])

# 前向填充
df = df.set_index('dt').asfreq('D', method='ffill').reset_index()

# 线性插值
df[['open', 'close', 'high', 'low']] = df[['open', 'close', 'high', 'low']].interpolate()
```

### Q2: 如何处理停牌数据？

A: 识别停牌并跳过处理：

```python
# 检测停牌（成交量为0）
suspended = df[df['vol'] == 0].copy()

# 策略中跳过停牌日
def should_trade(dt, suspended_dates):
    if dt in suspended_dates:
        return False
    return True
```

### Q3: 数据速度慢怎么办？

A: 使用缓存和批量请求：

```python
# 1. 启用缓存
dc = DataClient(token=token, cache_path="./cache")

# 2. 批量获取
symbols = ['000001', '000002', '600000']
for symbol in symbols:
    df = dc.get_kline(symbol, '日线', '20230101', '20231231')
    # 第一次慢，后续快（使用缓存）
```
